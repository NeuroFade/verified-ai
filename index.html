import { AttestationRegistry, ZKVerifier } from '@verified-ai/sdk'; // Initialize const registry = await AttestationRegistry.create(providerAddr); const verifier = await ZKVerifier.create(); // Verify inference const valid = await verifier.verifyProof(attestationId); console.log(valid ?
